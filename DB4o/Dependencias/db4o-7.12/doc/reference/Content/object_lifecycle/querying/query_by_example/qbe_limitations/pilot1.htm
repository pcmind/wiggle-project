package com.db4odoc.qbe;

public class Pilot1  {
  private String name;

  private int points;

  public Pilot1(String name)  {
    this.name = name;
    this.points = 100;
  }
  
  public String getName()  {
    return name;
  }
  
  public void setName(String name)  {
    this.name = name;
  }

  public int getPoints()  {
    return points;
  }

  public String toString()  {
    return name + "/" + points;
  }

}
